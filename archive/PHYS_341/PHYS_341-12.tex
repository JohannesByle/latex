\[
\begin{bmatrix}
C_1 \\
C_2
\end{bmatrix}
=
\begin{bmatrix}
1&1 \\
-(\beta-\sqrt{\beta^2-\omega_0^2})&-(\beta+\sqrt{\beta^2-\omega_0^2})
\end{bmatrix}^{-1}
\begin{bmatrix}
x_0 \\
v_0
